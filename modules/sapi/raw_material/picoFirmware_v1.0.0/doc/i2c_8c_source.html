80 void picoI2CInit(picoI2CPort p)
81 {
82  LPC_I2C_T * i2c = getI2CPort(p);
83  if (i2c) {
84  switch(p) {
85  case I2C0:
86  picoConfigI2C(P8_3);
87  picoConfigI2C(P8_6);
88  break;
89  case I2C1:
90  picoConfigI2C(P8_5);
91  picoConfigI2C(P8_8);
92  break;
93  case I2C2:
94  picoConfigI2C(P8_7);
95  picoConfigI2C(P2_13);
96  break;
97  default:
98  break;
99  }
100  Chip_I2C_Init(i2c);
101  Chip_I2CM_Enable(i2c);
102  Chip_I2C_SetClockDiv(i2c, I2C_CLK_DIVIDER);
103  Chip_I2CM_SetBusSpeed(i2c, I2C_BITRATE);
104  }
105 }

107 int32_t picoI2CTransfer(picoI2CPort p, uint8_t slave_addr,
108  uint8_t * txbuf, uint16_t txlen,
109  uint8_t * rxbuf, uint16_t rxlen)
110 {
111  int32_t rv = -1;
112  I2CM_XFER_T xfer;
113  LPC_I2C_T * i2c = getI2CPort(p);
114 
115  if (i2c) {
116  xfer.rxBuff = rxbuf;
117  xfer.rxSz = rxlen;
118  xfer.slaveAddr = slave_addr;
119  xfer.status = 0;
120  xfer.txBuff = txbuf;
121  xfer.txSz = txlen;
122 
123  Chip_I2CM_XferBlocking(i2c, &xfer);
124 
125  rv = xfer.status;
126  }
127 
128  return rv;
129 }
