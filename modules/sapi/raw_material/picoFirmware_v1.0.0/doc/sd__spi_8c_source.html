62 int32_t picoSDInit(void)
63 {
64  FRESULT r;
65  int32_t rv = -1;
66 
67  /* SPI configuration */
68  Chip_IOCON_PinMuxSet(LPC_IOCON, 0, 11, (IOCON_FUNC1 | IOCON_MODE_PULLUP | IOCON_DIGITAL_EN)); /* SPI0_SCK */
69  Chip_IOCON_PinMuxSet(LPC_IOCON, 0, 12, (IOCON_FUNC1 | IOCON_MODE_PULLUP | IOCON_DIGITAL_EN)); /* SPI0_MOSI */
70  Chip_IOCON_PinMuxSet(LPC_IOCON, 0, 13, (IOCON_FUNC1 | IOCON_MODE_PULLUP | IOCON_DIGITAL_EN)); /* SPI0_MISO */
71 #if 0
72  Chip_IOCON_PinMuxSet(LPC_IOCON, 0, 14, (IOCON_FUNC1 | IOCON_MODE_PULLUP | IOCON_DIGITAL_EN)); /* SPI0_SSEL0 */
73 #else
74  Chip_IOCON_PinMuxSet(LPC_IOCON, 0, 14, (IOCON_FUNC0 | IOCON_MODE_PULLUP | IOCON_DIGITAL_EN)); /* SPI0_SSEL0, GPIO */
75 #endif
76  //SPI init
77  SPI_CFGSETUP_T cfg;
78  SPIM_DELAY_CONFIG_T dcfg;
79 
80  Chip_SPI_Init(LPC_SPI0);
81  Chip_SPI_Enable(LPC_SPI0);
82 
83  cfg.lsbFirst = 0;
84  cfg.master = 1;
85  cfg.mode = SPI_CLOCK_MODE0;
86  Chip_SPI_ConfigureSPI(LPC_SPI0, &cfg);
87 
88  Chip_GPIO_SetPinDIROutput(LPC_GPIO, 0, 14);
89  Chip_GPIO_SetPinOutHigh(LPC_GPIO, 0, 14);
90 
91  Chip_SPIM_SetClockRate(LPC_SPI0, 100000);
92 
93  dcfg.PreDelay = 0;
94  dcfg.PostDelay = 0;
95  dcfg.FrameDelay = 0;
96  dcfg.TransferDelay = 0;
97  Chip_SPIM_DelayConfig(LPC_SPI0, &dcfg);
98 
99  Chip_SPIM_DisableLoopBack(LPC_SPI0);
100 
101  Chip_SPI_EnableInts(LPC_SPI0, (SPI_INTENSET_RXDYEN |
102  SPI_INTENSET_RXOVEN | SPI_INTENSET_TXUREN | SPI_INTENSET_SSAEN |
103  SPI_INTENSET_SSDEN));
104  NVIC_EnableIRQ(SPI0_IRQn);
105 
106  rtc_initialize();
107 
108  r = f_mount(&fs, "", 0);
109  if (r == FR_OK) {
110  rv = 0;
111  }
112  return rv;
113 }

169 picoFile picoSDOpen(char * filename, picoFileAccess fileAccess)
170 {
171  FIL * fp = (FIL *)malloc(sizeof(FIL));
172  BYTE mode;
173  if (fp) {
174  switch (fileAccess) {
175  case picoFILE_WRITE:
176  mode = FA_CREATE_ALWAYS | FA_WRITE;
177  break;
178  case picoFILE_APPEND:
179  mode = FA_OPEN_APPEND | FA_WRITE;
180  case picoFILE_READ:
181  default:
182  mode = FA_READ;
183  break;
184  }
185  FRESULT r = f_open(fp, filename, mode);
186  if (r != FR_OK) {
187  free(fp);
188  fp = NULL;
189  }
190  }
191  return (picoFile)fp;
192 }
