125 void picoUARTInit(picoUART_e u, picoPin_e txpin, picoPin_e rxpin, uint32_t baudrate)
126 {
127  LPC_USART_T * pu;
128  uint32_t irqn;
129 
130  switch (u) {
131  case UART1:
132  pu = LPC_USART1;
133  irqn = UART1_IRQn;
134  break;
135  case UART2:
136  pu = LPC_USART2;
137  irqn = UART2_IRQn;
138  break;
139  case UART3:
140  pu = LPC_USART3;
141  irqn = UART3_IRQn;
142  break;
143  default:
144  return;
145  }
146  picoConfigUART(txpin);
147  picoConfigUART(rxpin);
148  Chip_UART_Init(pu);
149  Chip_UART_ConfigData(pu, UART_CFG_DATALEN_8 | UART_CFG_STOPLEN_1 | UART_CFG_PARITY_NONE);
150  Chip_UART_Enable(pu);
151  Chip_UART_TXEnable(pu);
152  Chip_UART_SetBaud(pu, baudrate);
153 
154  RingBuffer_Init(&txrb[u], txbuff[u], 1, UART_BUFFER_LEN, NULL);
155  RingBuffer_Init(&rxrb[u], rxbuff[u], 1, UART_BUFFER_LEN, NULL);
156 
157  Chip_UART_ClearStatus(
158  pu,
159  UART_STAT_OVERRUNINT | UART_STAT_DELTARXBRK | UART_STAT_FRM_ERRINT | UART_STAT_PAR_ERRINT |
160  UART_STAT_RXNOISEINT);
161 
162  Chip_UART_IntEnable(pu, UART_INTEN_RXRDY);
163  Chip_UART_IntDisable(pu, UART_INTEN_TXRDY);
164 
165  NVIC_EnableIRQ(irqn);
166 }
