424 void picoAttachPinInterrupt(picoPin_e pin, picoPinIntRequest_e irq, picoPinIntEdgeType_e edge, picoPinIntHandler handler)
425 {
426  picoConfigDigitalInput(pin);
427  picoConfigPinInt(pin, irq);
428 
429  userHandlers[irq] = handler;
430 
431  Chip_PININT_SetPinModeEdge(LPC_PININT, 1<<irq);
432  if (PICOPININT_EDGE_FALLING == edge) {
433  Chip_PININT_EnableIntLow(LPC_PININT, 1<<irq);
434  } else {
435  Chip_PININT_EnableIntHigh(LPC_PININT, 1<<irq);
436  }
437  Chip_PININT_ClearIntStatus(LPC_PININT, 1<<irq);
438  NVIC_EnableIRQ(irqn[irq]);
439 }

441 void picoDetachPinInterrupt(picoPinIntRequest_e irq)
442 {
443  Chip_PININT_ClearIntStatus(LPC_PININT, 1<<irq);
444  NVIC_DisableIRQ(irqn[irq]);
445  Chip_PININT_DisableIntHigh(LPC_PININT, 1<<irq);
446  Chip_PININT_DisableIntLow(LPC_PININT, 1<<irq);
447  userHandlers[irq] = NULL;
448 }
