71 void picoSPIInit(picoSPIPort p, uint32_t clock_rate)
72 {
73  LPC_SPI_T * spi = getSPIPort(p);
74 
75  if (spi) {
76  switch (p) {
77  case SPI0:
78  picoConfigSPI(P6_11); /* SPI0_MOSI */
79  picoConfigSPI(P6_12); /* SPI0_SCK */
80  picoConfigSPI(P6_14); /* SPI0_MISO */
81  break;
82  case SPI1:
83  picoConfigSPI(P6_15); /* SPI1_MOSI */
84  picoConfigSPI(P6_16); /* SPI1_SCK */
85  picoConfigSPI(P6_18); /* SPI1_MISO */
86  break;
87  default:
88  break;
89  }
90  //SPI init
91  SPI_CFGSETUP_T cfg;
92  SPIM_DELAY_CONFIG_T dcfg;
93 
94  Chip_SPI_Init(spi);
95  Chip_SPI_Enable(spi);
96 
97  cfg.lsbFirst = 0;
98  cfg.master = 1;
99  cfg.mode = SPI_CLOCK_MODE0;
100  Chip_SPI_ConfigureSPI(spi, &cfg);
101 
102  Chip_SPIM_SetClockRate(spi, clock_rate);
103 
104  dcfg.PreDelay = 0;
105  dcfg.PostDelay = 0;
106  dcfg.FrameDelay = 0;
107  dcfg.TransferDelay = 0;
108  Chip_SPIM_DelayConfig(spi, &dcfg);
109 
110  Chip_SPIM_DisableLoopBack(spi);
111  }
112 }

114 int32_t picoSPITransfer(picoSPIPort p, picoPin_e ssel,
115  uint8_t * txbuf, uint8_t * rxbuf, int32_t len)
116 {
117  SPIM_XFER_T xfer;
118  int32_t rv = -1;
119  LPC_SPI_T * spi = getSPIPort(p);
120  if (spi) {
121  if (ssel != PICOPIN_COUNT) {
122  picoDigitalWrite(ssel, 0);
123  pauseus(10);
124  }
125  xfer.cbFunc = 0;
126  xfer.options = SPIM_XFER_OPTION_SIZE(8);
127  xfer.rxBuff = rxbuf;
128  xfer.txBuff = txbuf;
129  xfer.rxCount = xfer.txCount = len;
130  xfer.rxDoneCount = xfer.txDoneCount = 0;
131  xfer.sselNum = 0;
132  xfer.state = 0;
133  Chip_SPI_FlushFifos(spi);
134  Chip_SPIM_XferBlocking(spi, &xfer);
135  if (ssel != PICOPIN_COUNT) {
136  pauseus(10);
137  picoDigitalWrite(ssel, 1);
138  }
139  rv = xfer.state;
140  }
141  return rv;
142 }
