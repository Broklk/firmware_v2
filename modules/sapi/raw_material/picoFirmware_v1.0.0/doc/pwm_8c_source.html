71 static uint8_t getSCTch(picoPin_e pin)
72 {
73  uint8_t ch = 0xFF;
74  if ((pin < PICOPIN_COUNT) && (pin != P8_9)) {
75  switch (pin) {
76  case P6_5:
77  case P6_17:
78  ch = 0;
79  break;
80  case P6_7:
81  case P6_20:
82  ch = 1;
83  break;
84  case P6_10:
85  ch = 2;
86  break;
87  case P6_9:
88  ch = 3;
89  break;
90  case P2_1:
91  case P2_10:
92  case P6_14:
93  ch = 4;
94  break;
95  case P2_9:
96  case P6_13:
97  case P8_2:
98  ch = 5;
99  break;
100  case P2_8:
101  case P6_3:
102  ch = 6;
103  break;
104  case P2_7:
105  case P6_18:
106  ch = 7;
107  default:
108  ch = 0xFF;
109  break;
110  }
111  }
112  return ch;
113 }

124 int32_t picoPWMSet(picoPin_e pin, uint32_t duty)
125 {
126  uint8_t ch;
127  int32_t rv = -1;
128 
129  if ((duty <= 100) && (picoConfigSCTOutput(pin) == 0)) {
130  ch = getSCTch(pin);
131  if (ch != 0xFF) {
132  Chip_SCTPWM_Stop(LPC_SCT);
133  Chip_SCTPWM_SetOutPin(LPC_SCT, ch, ch);
134  Chip_SCTPWM_SetDutyCycle(LPC_SCT, ch,
135  Chip_SCTPWM_PercentageToTicks(LPC_SCT, duty));
136  Chip_SCTPWM_Start(LPC_SCT);
137  rv = 0;
138  }
139  }
140  return rv;
141 }
